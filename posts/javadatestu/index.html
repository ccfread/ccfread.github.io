package com.ezxmt.collection;

import com.ezxmt.hook.Cat;
import com.ezxmt.hook.Dog;

import java.util.Date;

/**
 * 学习ArrayList,
 */
public class TestArrayList {

    //创建obj数组，用于内部存放数据
    private Object[] elementDate;

    //大小
    private int size;

    //构造方法
    public TestArrayList(){
        //不指定大小时默认给10
        this(10);
    }

    public TestArrayList(int initSize){
        if(initSize<0){
            //数组长度小于0，抛出异常
            try {
                throw new Exception("数组长度小于0");
            } catch (Exception e) {
                throw new RuntimeException(e);
            }
        }
        //初始化数组
        elementDate = new Object[initSize];

    }

    public void add(Object obj){
        //判断当前数组是否需要扩容
        if(size>=elementDate.length){
            Object[] newArray = new Object[size*2+1];
            //拷贝数组
            System.arraycopy(elementDate,0,newArray,0,elementDate.length);
            //或者for循环读取
           /*for (int i = 0; i < elementDate.length; i++) {
                newArray[i] = elementDate[i];
            }*/
            elementDate = newArray;
        }

        //两种写法
        /* 第一种写法
        elementDate[size] = obj;
        size++;
         */
        //第二种写法，先处理，后自增
        elementDate[size++]=obj;
    }

    //根据下标获取数组对应的元素
    public Object get(int index){
        //首先判断下标是否在数组长度范围内
        if(index<0 || index>=size){
            try {
                throw new Exception("数组下标输入错误");
            } catch (Exception e) {
                throw new RuntimeException(e);
            }
        }
        //根据下标获取对象并返回
        return elementDate[index];
    }

    public int getSize() {
        return size;
    }

    //移除指定下标的对象
    public void remove(int index){

    }

    public static void main(String[] args) {
        TestArrayList t = new TestArrayList();
        t.add(123);
        t.add("912hj98g2");
        t.add(new Dog());
        t.add(new Cat());
        t.add(new Date());
        t.add("abc");
        t.add("bcd");
        t.add("cdf");
        t.add("alskdjf");
        t.add(new Dog());
        t.add("ddd");
        t.add("bbb");

        System.out.println(t.getSize());
        System.out.println(t.get(12));
    }
}
