HashMap使用，统计单词出现的次数，基础版，写的不好

package collenction;

import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;
import java.util.Set;

/**
 *  this is a cat and that is a mice and where is the foot?
 * 统计每个单词出现的次数
 *
 *存储到Map中
 * key：String
 * value：自定义类型
 *  
 *  @author Administrator
 */
public class HashMapTest01 {

	public static void main(String[] args) {
		//定义字符串
		String str = "this is a cat and that is a mice and where is the foot?";
		//分割字符串存储到Map中，根据空格分割
		String[] strArray = str.split(" ");
		
		Map<String,Letter> letter = new HashMap<String,Letter>(); 
		
		/*
		 * 实现方式一
		 */
	
		for(String temp:strArray) {
			//	为所有的key创建容器
			if(!letter.containsKey(temp)) {
				letter.put(temp,new Letter());				
			}
			//之后容器中存放对应value
			Letter col = letter.get(temp);
			col.setCount(col.getCount()+1);
		}
		
		//输出Map的值
		Set<String> keys = letter.keySet();
		Iterator i = keys.iterator();
		while (i.hasNext()) {
			Letter col = letter.get(i.next()	);
			System.out.println("字母"+i.next()	+"出现了"+col.getCount()+"次");
		}
		 
		for (String key : keys) {
			Letter col = letter.get(key);
			System.out.println("字母"+ key	+"出现了"+col.getCount()+"次");

		}
		System.out.println("************");
		/*
		 * 实现方式二
		 */
		Letter col = null;
		for(String temp:strArray) {
			//	判断容器是否存在
			if((col = letter.get(temp))==null) {
				//不存在就创建一个
				letter.put(temp,new Letter());				
			}
			//之后容器中存放对应value
			col.setCount(col.getCount()+1);
		}
		
		
		//输出Map的值
				Set<String> keyss = letter.keySet();
				for (String key : keyss) {
					col = letter.get(key);
					System.out.println("字母"+ key	+"出现了"+col.getCount()+"次");
				}
		
    /*
    * 当前Letter的name没用上，也可以存储key
    */
	}
}

创建实体类Letter用于存储出现次数

package collenction;

public class Letter {

	private String name;
	private int count;
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public int getCount() {
		return count;
	}
	public void setCount(int count) {
		this.count = count;
	}
	public Letter(String name, int count) {
		super();
		this.name = name;
		this.count = count;
	}
	public Letter() {
	
	}
	
}
